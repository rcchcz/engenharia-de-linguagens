void mdc(int n, int m, int *r){
    if(n % m == 0) {
        *r = m;
    } else if(m % n == 0){
        *r = n;
    } else if(n < m){
        int resA = m % n;
        mdc(n,resA,r);
    } else {
        int resB = n % m;
        mdc(m, resB, r);
    }
}

    int init = 0;
